let fibs =
    let generator state =
        let (cur, prev) = state
        let next = cur + prev
        let state' = (next, cur)
        Some (cur, state')
    let more =
        (1, 0)
        |> Seq.unfold generator
    seq {
        0
        yield! more
    }

let first20 = Seq.take 20 fibs
